"""IP range scanning tool

from scans.port_scan.service_scan import service_scan
from report.report import netscan_report
from report.print_selfscan import selfscan_report
from custom.threads.CustomThread import THREAD_DONE
from parsing.port_parser import Port_Parse
from visualization.visualizer import visualizer
from scans.os_scan.os_detection import (detect_os_by_ttl,
                         detect_os_by_window,
                         detect_os_by_ttl_df,
                         detect_os_tcp_options)

def main(args: argparse.Namespace, parser: argparse.ArgumentParser) -> None:
    """The main function for argparse.py

    Args:
        args (argparse.Namespace): The command-line arguments passed into main
        parse (argparse.ArgumentParser): The command-line argument parser
    """
    os.makedirs("output", exist_ok=True)

    if args.ip is None and not args.selfscan and not args.list:
        parser.print_help()
    
    if args.list:
        list_scanner = Scan(None, None, None)
        list_scanner.list_plugins()
        sys.exit(0)
    
    try:
        if args.service_scan: # Makes sure that the portscan runs if they chose a service scan      
            if args.portscan is None:
                args.portscan = ""

        if "all" not in args.top_ports:
            args.top_ports = int(args.top_ports)
            
        #perform optional self scan
        if args.selfscan:
            results = scan_self()
            selfscan_report(results)
    
        #perform primary network scan
        if args.ip is not None:
            network_map = scan_network(args)
            
            #perform optional port scan
            if args.portscan is not None:
                if args.portscan == "" and not args.top_ports:
                    ports = ""
                else:
                    port_parser = Port_Parse()
                    ports = port_parser.port_parse(args.portscan, args.top_ports)
            
                scan_ports(network_map, args.timeout, ports)
            
            if args.service_scan:
                service_scan(network_map, args.service_scan_timeout)
                
            #perform optional os detection
            if args.os:
                detect_os_by_ttl(network_map)
                detect_os_by_window(network_map)
                detect_os_by_ttl_df(network_map)
                detect_os_tcp_options(network_map)
                
            #report findings
            netscan_report(network_map, args)
            
            if args.traceroute: 
                # Only visualize if a traceroute was performed
                # Otherwise you get a single 127.0.0.1 node in the graph
                visualizer(network_map)
            
            #quick visualization of network_map structure
            if args.debug:
                print("\n[Network Map Contents]")
                for ip_str, addr_info in network_map.items():
                    print(f"  IP: {ip_str}")
                    print(f"    Object IP: {addr_info.ip()}")
                    print(f"    Open Ports: {addr_info.oports}")
                    print(f"    Closed Ports: {addr_info.cports}")
                    print(f"    Hops: {addr_info.hops}")

                import pickle # pickling the state after the run to look at later
                from datetime import datetime
                date = datetime.now().strftime("%Y-%m-%d-%H-%M-%S")
                with open(f"output/{date}debug.pickle", "wb") as out:
                    pickle.dump(network_map, out, protocol=pickle.HIGHEST_PROTOCOL)  
            
    except KeyboardInterrupt:
        print("\n\nExiting program...")
        THREAD_DONE.set()
        sys.exit(0)
    except Exception as e: # If any other exception occurs, make sure it ends the thread
        print(e) 
        THREAD_DONE.set()
        sys.exit(0)



if __name__ == "__main__":
    #parse args
    parser = argparse.ArgumentParser(usage='%(prog)s [options]')
    parser.add_argument('-i', '--ip', help="The IP range to scan")
    parser.add_argument('-p', '--protocol',
    		help='The type of scan to perform. Default "arp"')
    parser.add_argument('-l', '--list', action="store_true",
                        help="Lists the valid protocols to use, and exits.")
    parser.add_argument('-s', '--selfscan', action='store_true',
    		help="Scans the user's /proc system")
    parser.add_argument('-t', '--timeout', default=2, type=float,
    		help='The time in seconds to wait for a packet. Default 2 seconds')
    parser.add_argument('-ps', '--portscan', nargs='?', const="",
        default=None, type=str,
        help='Perform a portscan. Specify ports or leave blank for common ports.')
    parser.add_argument('-tr', '--traceroute', choices=['icmp', 'tcp'],
			help='Perform traceroute on discovered hosts. Options: icmp, tcp')
    parser.add_argument('-trh', '--traceroutehops', default=30, type=int,
            help='Max number of hops for traceroute')
    parser.add_argument('-v', '--inverted-match', default=None,
            help='Takes in a file of IP addresses to NOT scan')
    parser.add_argument('-d', '--debug', action='store_true',
    		help="Show extra debug info, and creates a debug pickle")
    parser.add_argument("--top-ports", default="0", choices=["all", "0", "10", "25", "100", "1000"],
            help="Scan the top x ports on every machine. Options: all, 10, 25, 100, 1000")
    parser.add_argument("-sV", '--service-scan', action="store_true",
                        help="Attempts to Identify the service running on each port")
    parser.add_argument('-sVt', '--service-scan-timeout', default=5, type=float,
    		help='The time in seconds to wait for a packet. Default 5 seconds')
    parser.add_argument('-c', '--cve', action='store_true',
            help='Check found services for known vulnerabilities.')
    parser.add_argument('-o', '--os', action='store_true',
            help='Try to determine the OS of found hosts.')
    
    #send args to main
    args = parser.parse_args()
    main(args, parser)
