def analyze(args: argparse.Namespace) -> None:
    """The function to analyze the given network

    Args:
        args (argparse.Namespace): The command-line arguments passed into main

    Raises:
        ValueError: If the passed in IP address range is not valid
    """
    term_size = os.get_terminal_size().columns
    cols = term_size // 17 - 2

    normalized = normalize_cidr(args.ip)
    if not normalized:
        raise ValueError(f"{args.ip} is not a valid IPv4 range")

    scan_type = args.scan if args.scan else "arp"
    timeout = int(args.timeout) if args.timeout else 2

    scanner = Scan(normalized, timeout=timeout, scan_type=scan_type)
    results = scanner.scan()

    while len(results) % cols != 0:
        results.append('') # Padding out for the final line

    print("IP Addresses That Responded:")
    print("+-----------------" * cols + "+")

    for i in range(len(results) // cols):
        # Creates the rows of IP addresses
        row = [results[x] for x in range(i * cols, i * cols + cols)]

        # Dislays the rows in a nice table
        print("| " + " | ".join(f"{j:<15}" for j in row) + " |")
        print("+-----------------" * cols + "+")

def main(args: argparse.Namespace, parser: argparse.ArgumentParser) -> None:
    """The main function for argparse.py

    Args:
        args (argparse.Namespace): The command-line arguments passed into main
        parse (argparse.ArgumentParser): The command-line argument parser
    """
    if args.ip is not None:
        analyze(args)
    else:
        parser.print_help()



if __name__ == "__main__":
    #parse args
    parser = argparse.ArgumentParser(usage='%(prog)s -i [IP] [options]')
    parser.add_argument('-i', '--ip', help="The IP range to scan")
    parser.add_argument('-s', '--scan', help='The type of scan to perform. Default "arp"')
    parser.add_argument('-t', '--timeout',
                        help='The time in seconds to wait for a packet. Default 2 seconds')

    #send args to main
    args = parser.parse_args()
    main(args, parser)
