#! /usr/bin/env python

"""IP range scanning tool
"""

import os
import argparse
from netscan import scan_network
from selfscan import scan_self
from report import netscan_report, selfscan_report

def main(args: argparse.Namespace, parser: argparse.ArgumentParser) -> None:
    """The main function for argparse.py

    Args:
        args (argparse.Namespace): The command-line arguments passed into main
        parse (argparse.ArgumentParser): The command-line argument parser
    """
    
    if args.ip is None and not args.selfscan:
        parser.print_help()
    
    if args.ip is not None:
        results = scan_network(args)
        netscan_report(results, args.protocol, args.traceroute, args.timeout)
        
    if args.selfscan:
        results = scan_self()
        selfscan_report(results)



if __name__ == "__main__":
    #parse args
    parser = argparse.ArgumentParser(usage='%(prog)s [options]')
    parser.add_argument('-i', '--ip', help="The IP range to scan")
    parser.add_argument('-p', '--protocol',
    		help='The type of scan to perform. Default "arp"')
    parser.add_argument('-s', '--selfscan', action='store_true',
    		help="Scans the user's /proc system")
    parser.add_argument('-t', '--timeout',
    		help='The time in seconds to wait for a packet. Default 2 seconds')
    parser.add_argument('-tr', '--traceroute', choices=['icmp', 'tcp'],
			help='Perform traceroute on discovered hosts. Options: icmp, tcp')
                    
    #send args to main
    args = parser.parse_args()
    main(args, parser)
