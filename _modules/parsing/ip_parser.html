def normalize_cidr(ip_input: str) -> str:
    """Takes an IP address, range of IP addresses, or partial IP address
    and normalizes it to CIDR notation.

    Args:
        ip_input (str): A string representing an IPv4 address range

    Returns:
        str: A string representing the IP range in CIDR notation
    """
    try:
        # Written by AI
        # Prompt: Write a line in python that validates an IP address range
        network = ipaddress.ip_network(ip_input, strict=False)
        return str(network)

    except ValueError:
        pass

    parts = ip_input.split('.')

    while len(parts) < 4:
        parts.append('0')

    for part in parts:
        try:
            octet = int(part)

            if octet < 0 or octet > 255:
                return None

        except ValueError:
            return None

    original_parts = len(ip_input.split('.'))
    suffix_map = { # For CIDR notation
        1: 8,
        2: 16,
        3: 24,
        4: 32
    }

    suffix = suffix_map.get(original_parts, 32)

    full_ip = '.'.join(parts)

    try:
        network = ipaddress.ip_network(f"{full_ip}/{suffix}", strict = False)
        return str(network)
    except ValueError:
        return None

def remove_ips(file: str, ips: list[ipaddress.IPv4Address]) -> list[ipaddress.IPv4Address]:
    """Removes the IP addresses in the file given from the list of IP addresses.

    Args:
        file (str): The path to the file of exclusions
        ips (list[ipaddress.IPv4Address]): The list containing IPv4 addresses

    Returns:
        (list[ipaddress.IPv4Address]): A list of the IP addresses with the chosen removed
    """
    try:
        with open(file, 'r') as inp:
            lines = inp.readlines()


        for i in range(len(lines)):
            cidr = normalize_cidr(lines[i].strip())
            ip_addresses = list(ipaddress.ip_network(cidr).hosts())
            for host in ip_addresses:
                ips.remove(host)
        return ips


    except:
        return ips
