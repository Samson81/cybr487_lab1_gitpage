class Port_Parse:
    """The object for parsing the common services file given by NMAP
    """
    all = {}
    thousand = {}
    hundred = {}
    twentyfive = {}
    ten = {}

    def __init__(self) -> None:
        """Initializes the Port_Parse object
        """
        try:
            service_files = os.listdir('utilities/services')
            if len(service_files) < 5: # Make sure all files are there
                self._get_ports()
        except OSError:
            self._get_ports()
            service_files = os.listdir('utilities/services')
        
        # Fill all dictionaries globally
        Port_Parse.all = self._read_ports()
        Port_Parse.thousand = self._read_ports(1000)
        Port_Parse.hundred = self._read_ports(100)
        Port_Parse.twentyfive = self._read_ports(25)
        Port_Parse.ten = self._read_ports(10)

    def _get_ports(self) -> None:
        """Reads in the list of TCP ports and services from the 
        nmap-services file on the NMAP GitHub repository.

        If the service/ directory does not exist, this will get run run, otherwise 
        they will be read in from the files.
        """
        try:
            temp = requests.get(LINK).text # Gets the list of ports
        except:
            print(f"Cannot get {LINK}, sources cannot be populated.")
        ports = {}
        ports_temp = {}

        for line in temp.split('\n'):
            if not line.startswith("#") and "tcp" in line: # Ignores comments and UDP lines
                port = line.split()
                port[1] = int(port[1][:port[1].rfind("/")]) # The port number
                port[2] = float(port[2]) # The frequency

                ports[port[1]] = Port(port[0].upper(), port[2]) # port # = Port(service, frequency)
                ports_temp[port[1]] = {'service': port[0].upper(),
                                        'frequency': port[2]} # This is soley for outputting to a file
            
        
        os.makedirs('utilities/services', exist_ok=True)
        
        top_list = []

        for jkey in ports_temp.keys():
            top_list.append((jkey, ports_temp[jkey]['frequency'])) # Append (port #, frequency) tuple
        
        top_list = sorted(top_list, key=lambda freq: freq[1], reverse=True) # Sort by frequency

        for i in ['', 1000, 100, 25, 10]:
            ports_json = {}

            if i:
                top_list = top_list[:i] # Only grab the top i ports
            
            top_ports = []
            for p in top_list:
                top_ports.append(p[0]) # Grab the port numbers
            
            for jkey in ports_temp.keys():
                if jkey in top_ports:
                    ports_json[jkey] = ports_temp[jkey] # Add keys to a new dictionary that are in the top_ports list

                
            out_json = json.dumps(ports_json, sort_keys=True, indent=2)

            with open(f'utilities/services/{i}services.json', "w") as out:
                out.write(out_json) # output to a file
        
    def _read_ports(self, top: int = -1) -> dict[int, Port]:
        """Reads in a service file, and converts it to the right format for reading.

        Args:
            top (int = -1): The number on the file to read in. If -1, then it reads 
            in services.json. Otherwise, it reads in <top>services.json. Default -1

        Returns:
            (dict[int, Port]): A dictionary with the port number as the key, and the service and
            frequency put into a custom object as the data.
        """
        file = ""
        if top > 0:
            file = str(top)
        
        with open(f'utilities/services/{file}services.json', "r") as inp:
            data = json.loads(inp.read())
        
        output_dict = {}
        for key in data.keys(): # Read in data into custom object
            output_dict[int(key)] = Port(data[key]['service'], data[key]['frequency'])
        
        return output_dict

    def port_parse(self, ports: str, top: int = 0) -> set[int]:
        """Takes in a string of port ranges and create a list of all ports to search. 
        If top is a specified value other than 0, than it scans the following:

        +-------+----------+
        | Value | # Ports  |
        +=======+==========+
        | 10    | Top 10   |
        +-------+----------+
        | 25    | Top 25   |
        +-------+----------+
        | 100   | Top 100  |
        +-------+----------+
        | 1000  | Top 1000 |
        +-------+----------+

        **Note**: Based on nmaps "nmap-services" file

        Args:
            ports (str): The range of ports to scan
            top (int): Whether or not to include top ports, and what number

        Returns:
            (set[int]): Contains all desired port numbers to scan
        """
        temp_list = []

        if isinstance(top, str): # Append the desired top amount
            temp_list += self.all.keys()
        elif top == 1000:
            temp_list += self.thousand.keys()
        elif top == 100:
            temp_list += self.hundred.keys()
        elif top == 25:
            temp_list += self.twentyfive.keys()
        elif top == 10:
            temp_list += self.ten.keys()

        ranges = ports.split(',') # Convert to list
        for i in ranges:
            if not i:
                continue
            if "-" in i:
                r1 = i.split('-') # Append x-y ranges from x to y inclusive
                for x in range(int(r1[0]), int(r1[-1]) + 1):
                    temp_list.append(x)
            else:
                temp_list.append(int(i))

        return set(temp_list)
