def netscan_report(ip_list: list[str],
                   protocol: str,
                   traceroute_type: str | None = None,
                   timeout: int = 2) -> None:
    """Display detailed information for each discovered IP
    
    Args:
        ip_list (list[str]): The list of IP addresses that were discovered
        protocol (str): The scan type that is being done
        traceroute_type (str | None): The type of traceroute. Default None
        timeout (int): The timeout for each packet. Default 2
    """
    
    for ip in ip_list:
        host_text = f"Host: {ip}"
        box_width = len(host_text) + 4
    
        print(f"\n┌{'─' * (box_width - 2)}┐")
        print(f"│ {host_text} │")
        print(f"└┬{'─' * (box_width - 3)}┘")
        print(f" │")
        
        #do the port scan
        prefix = ' ├─' if traceroute_type else ' └─'
        print(f"{prefix}[Port Scan]")
        port_results = scan_ports(ip, timeout)  #returns dict of open ports
        prefix = ' │   ' if traceroute_type else '     '
        for port, service in port_results.items():
            print(f"{prefix}Port {port}/tcp: OPEN ({service})")
        
        #do the traceroute if requested
        if traceroute_type:
            print(f" │")
            print(f" └─[Traceroute - {traceroute_type.upper()}]")
            hops = traceroute(ip, traceroute_type, timeout)  #returns list of hops
            for i, hop in enumerate(hops, 1):
                print(f"     {i:02}. {hop}")
                
    print()

def print_boxed_line(text: str, width: int, start: str = "   │ ") -> None:
    """Print text in a box with padding to match width

    Args:
        text (str): The text to put in the box
        width (int): The width of the box
        start (str): The beginning of the line 
    """
    padding = max(0, width - len(text))
    print(f"{start}{text}{' ' * padding} │")

def selfscan_report(listening_ports: list[dict]) -> None:
    """Generate a formatted report of listening ports

    Args:
        listening_ports (list[dict]): A list of the ports being listed on.
    """
    if not listening_ports:
        return "No listening ports found.\n"
    
    #sort by port number
    listening_ports.sort(key=lambda x: x['port'])

    #build header box
    header_text = f"Self Scan Report"
    header_width = len(header_text) + 4

    print(f"\n┌{'─' * (header_width - 2)}┐")
    print(f"│ {header_text} │")
    print(f"└┬{'─' * (header_width - 3)}┘")
    print(f" │")
    
    #build content box
    lines = []
    for port in listening_ports:
        pid_str = str(port['pid']) if port['pid'] else "N/A"
        process_str = port['process'] if port['process'] else "N/A"

        line = f"{port['port']:<8} {port['protocol']:<10} {port['address']:<18} {pid_str:<8} {process_str}"
        lines.append(line)

    #find the longest line
    content_width = max(len(line) for line in lines)

    print(f" │ ┌{'─' * content_width}──┐")

    #print column headers and seperator
    header = f"{'PORT':<8} {'PROTOCOL':<10} {'ADDRESS':<18} {'PID':<8} {'PROCESS'}"
    print_boxed_line(header, content_width, " │ │ ")
    print(f" └─┤{'-' * content_width}--│")

    #print each line
    for line in lines:
        print_boxed_line(line, content_width)
    
    print(f"   │{'-' * content_width}--│")
    
    total = f"Total listening ports: {len(listening_ports)}"
    print_boxed_line(total, content_width)

    print(f"   └{'─' * content_width}──┘\n")
