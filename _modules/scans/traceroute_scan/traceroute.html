def traceroute(target_info: Address_Info.Address_Info,
               args: Namespace) -> None:
    """Perform traceroute to target IP using ICMP or TCP
    
    Args:
        target_info: Address_Info object to populate with hop data
        args (argparse.Namespace): The arguments passed into the analyzer

    """
    icmp = args.traceroute.lower() == 'icmp'

    # Use the first open port in the list if there is one, otherwise default to 80
    if len(target_info.oports) < 1:
        port = 80
    else:
        port = target_info.oports[0]

    hops = _traceroute(target_info.ip(), 
                       args.timeout,
                       args.traceroutehops, 
                       icmp,
                       port=port) 
    
    for hop in hops:
        target_info.add_hop(hop[0], hop[1])



def _traceroute(target_ip: str,
                    timeout: int,
                    max_hops: int,
                    icmp: bool,
                    port: int = 80) -> list[str]:
    """Perform Either ICMP or TCP traceroute
    
    Uses ICMP Echo Request packets or TCP SYN packets with incrementing TTL values.
    Each hop is probed 3 times to handle packet loss.
    
    Args:
        target_ip (str): Destination IP address
        timeout (int): Timeout for each probe
        max_hops (int): Maximum hops to try
        icmp (bool): Whether the traceroute is ICMP (True) or TCP (False) packet
        port: (int = 80): The port to scan on for TCP
        
    Returns:
        list[str]: Formatted hop information
    """
    hops = []
    reached_destination = False

    for ttl in range(1, max_hops + 1):
        hop_info = probe_hop(target_ip, ttl, timeout, icmp, port)
            
        if hop_info['ip'] is None:
            hops.append((hop_info['ip_display'], hop_info['avg_rtt']))
        else:
            hops.append((hop_info['ip'], hop_info['avg_rtt']))
           
        #set color
        if hops[-1][1] == -1:
            color = RED
        elif hops[-1][1] > 100:
            color = YELLOW
        else:
            color = ''
            
        print(f"    │ {ttl:02}. {color}{hops[-1][0]} ", end="")
        if hops[-1][1] == -1:
            print(f"(Request Timed Out){RESET}")
        else:
            print(f"({hops[-1][1]:.2f} ms){RESET}")
            
        #cjeck if we reached the destination
        if hop_info['ip'] == target_ip:
            reached_destination = True
            break
    
    if not reached_destination and ttl == max_hops:
        last_hop = hops.pop(-1)
        hops.append((last_hop[0] + f"\n{target_ip}", last_hop[1]))# Adds the IP address to indicate that it reached max
        print(f"    │ {RED}Max hops ({max_hops}) reached - destination may be further{RESET}")
    
    return hops

def probe_hop(target_ip: str,
              ttl: int,
              timeout: int,
              icmp: bool,
              port: int = 80,
              attempts: int = 3) -> dict[str, str]:
    """Probe a single hop using ICMP or TCP packets, attempting 3 times
    
    Args:
        target_ip (str): Destination IP
        ttl (int): Time-to-live value for this hop
        timeout (int): Timeout for probe
        icmp (bool): Whether to send an ICMP packet (True) or TCP packet (False)
        port (int = 80): Target TCP port
        attempts (int = 3): Number of attempts per hop (default 3)
        
    Returns:
        (dict[str, str]): Contains 'ip', 'ip_display', 'avg_rtt', 'responses'
    """

    responses = []
    rtt_times = []
    responding_ip = None

    if icmp:
        make_packet = make_icmp
        value = random.randbytes(20) # value is used to pass in the ping data, or the port into tcp. Can be used to verify packets
    else:
        make_packet = make_tcp
        value = port

    for attempt in range(attempts):
        packet = make_packet(target_ip, ttl, value)
        
        #send packet and wait for response
        ans, unans = sr(packet, timeout=timeout, verbose=0)

        if ans:
            for sent, recv in ans:
                rtt = (recv.time - sent.sent_time) * 1000
                if icmp:
                    is_valid = recv.haslayer(ICMP) and (recv[ICMP].type == 0 or recv[ICMP].type == 11)
                else:
                    is_icmp = recv.haslayer(ICMP) and recv[ICMP].type == 11
                    is_valid = is_icmp or (recv.haslayer(TCP) and recv[TCP].flags == "SA")
                    
                responses.append(is_valid)

                if responding_ip is None and (is_valid):
                        responding_ip = recv[IP].src
                
                if is_valid:
                    rtt_times.append(rtt)  #milliseconds probably better
        else:
            responses.append(False)
    
    #calc results
    successful_probes = sum(responses)
    
    if successful_probes == 0:
        #all packets dropped
        return {
            'ip': None,
            'ip_display': '* * *',
            'avg_rtt': -1,
            'responses': responses
        }
    else:
        avg_rtt = sum(rtt_times) / len(rtt_times)
        return {
            'ip': responding_ip,
            'ip_display': responding_ip,
            'avg_rtt': avg_rtt,
            'responses': responses
        }

def make_icmp(target_ip: str, ttl: int, data: bytes = "") -> scapy.layers.inet.IP:
    """Creates and returns an ICMP Echo Request packet

    Args:
        target_ip (str): The IP address to target with Echo Request
        ttl (int): The ttl for the packet
        data (bytes = ""): The data to send in the Echo Request

    Returns:
        (scapy.layers.inet.IP): The ICMP Echo Request packet to send
    """
    ip_packet = IP()
    icmp_packet = ICMP()

    ip_packet.dst = target_ip
    ip_packet.ttl = ttl

    icmp_packet.type = 8 # Echo request

    return ip_packet/icmp_packet/data

def make_tcp(target_ip: str, ttl: int, port: int = 80) -> scapy.layers.inet.IP:
    """Creates and returns a TCP SYN packet

    Args:
        target_ip (str): The IP address to target with the SYN packet
        ttl (int): The ttl for the packet
        port (int = 80): The port to target with the SYN packet

    Returns:
        (scapy.layers.inet.IP): The TCP SYN packet to send
    """
    sport = random.randint(1024,65535)
    ip_packet = IP()
    tcp_packet = TCP()

    ip_packet.dst = target_ip
    ip_packet.ttl = ttl

    tcp_packet.dport = port
    tcp_packet.sport = sport
    tcp_packet.flags = "S"

    return ip_packet/tcp_packet
