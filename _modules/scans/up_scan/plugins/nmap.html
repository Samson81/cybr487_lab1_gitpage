def scan(ip: (str | list[str]),
         timeout: int = 2) -> list[str]:
    """A scan that mimics the default nmap scan.

    This scan sends the following packets to each IP address:
    - ARP 
    - ICMP Echo
    - ICMP Timestamp
    - TCP ACK on port 80
    - TCP SYN on port 443

    Args:
        ip (str | list[str]): A string or list that contain the IP addresses to scan
        timeout (int): The time in seconds to wait for a packet. Default 2
    
    Returns:
        list[str]: A list that contains all the IP addresses from the 5 scans
    """
    
    print("**Note**: This scan can take some time")

    proc = TaskRunner()

    proc.dispatch(arp.scan, ip=ip, timeout=timeout) # ARP
    proc.dispatch(icmp.scan, ip=ip, timeout=timeout) # Echo Request
    proc.dispatch(icmp.scan, ip=ip, timeout=timeout, flags=13) # Timestamp
    proc.dispatch(tcp.scan, ip=ip, timeout=timeout, port=80, flags="A") # TCP ACK
    proc.dispatch(tcp.scan, ip=ip, timeout=timeout, port=443, flags="S") # TCP SYN
    # Start threads
    
    results = proc.gather_results()

    seen_ip = []
    for result in results:
        seen_ip += result
    
    return seen_ip
