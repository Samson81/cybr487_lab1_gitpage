def detect_os_by_ttl(ip_dict: dict[str, Address_Info.Address_Info]) -> None:
    """Perform Operating System Detection through time-to-live (ttl) values, 
        then set the Operating System name and our amount of confidence of it
    
     Args: 
        ip_dict (dict[str, Address_Info.Address_Info]): The dictionary of IPs to parse through along with their respective hops
    """
    
    

    for ip_addr in ip_dict.keys():
        packet = IP(dst = ip_addr)/TCP(dport = 80, flags = "S")
        response = sr1(packet, timeout=2, verbose=0)
        
        if response and response.haslayer(TCP):
            ttl = response[IP].ttl
            if ttl <= 64:
                ip_dict[ip_addr].set_os_info("Linux/Unix", 70)
            
            elif ttl <= 128:
                 ip_dict[ip_addr].set_os_info("Windows", 70)
                
            
            elif ttl <= 255:
                ip_dict[ip_addr].set_os_info("Cisco/Network Device", 75)
                
            
            else:  ip_dict[ip_addr].set_os_info("Unknown", 0)

def detect_os_by_window(ip_dict: dict[str, Address_Info.Address_Info]) -> None:
    """Perform Operating System Detection through TCP window size
    
    Args: 
        ip_dict (dict[str, Address_Info.Address_Info]): The dictionary of IPs to parse through along with their respective hops
    """
   
    for ip_addr in ip_dict.keys():
      packet = IP(dst = ip_addr)/TCP(dport = 80, flags = "S")
      responses = sr1(packet, timeout=2, verbose=0)

      
      if responses and responses.haslayer(TCP):
            window_size = responses[TCP].window
            
            if window_size == 65535:
                ip_dict[ip_addr].set_os_info("Windows 10/11", 75)
                
            elif window_size == 8192:
                ip_dict[ip_addr].set_os_info("Windows 7/8", 75)

            elif window_size == 16384:
                ip_dict[ip_addr].set_os_info("Windows server", 75)
            
            elif window_size == 5840:
                ip_dict[ip_addr].set_os_info("Linux", 75)
            
            elif window_size == 5720:
                ip_dict[ip_addr].set_os_info("Linux", 75)

def detect_os_by_ttl_df(ip_dict: dict[str, Address_Info.Address_Info]) -> None:
    """Performs OS detection through TTL and the Do Not Fragment flag

    Args: 
        ip_dict (dict[str, Address_Info.Address_Info]): The dictionary of IPs to parse through along with their respective hops
    """

    for ip_addr in ip_dict.keys():
        packet = IP(dst = ip_addr)/TCP(dport = 80, flags = "S")
        response = sr1(packet, timeout=2, verbose=0)

        if response and response.haslayer(TCP):
            ttl = response[IP].ttl
            window = response[IP].window
            df_set = bool(response[IP].flags & 0x02) !=0

            if ttl <= 64 and df_set:
                ip_dict[ip_addr].set_os_info("Linux/Unix (Modern)", 80)
            
            elif ttl <=64 and df_set==False:
                ip_dict[ip_addr].set_os_info("Linux/Unix (Old)", 75)
            
            elif ttl <= 128 and df_set:
               ip_dict[ip_addr].set_os_info("Windows 10/11", 80)
            
            elif ttl <= 128 and df_set==False:
                ip_dict[ip_addr].set_os_info("Windows 7/8 or older", 75)
            
            elif ttl <= 255:
                ip_dict[ip_addr].set_os_info("Cisco/Network Device", 75)
            
            elif ttl <= 64 and window == 5840 and df_set:
                ip_dict[ip_addr].set_os_info("Linux 2.6.x-5.x", 85)

def detect_os_tcp_options(ip_dict: dict[str, Address_Info.Address_Info]) -> None:
    """Perform Operating System Detection through finding the IPs TCP fingerprint
    
    Args: 
        ip_dict (dict[str, Address_Info.Address_Info]): The dictionary of IPs to parse through along with their respective hops
    """
    TCP_OPTION_FINGERPRINTS = {
    # Linux variants
    "M1460NNTSTW7": ("Linux 3.x-5.x", 90),
    "M1460NNTSTW8": ("Linux 5.x+", 90),
    "M1460ST00NNW7": ("Linux 2.6.x", 85),
    
    # Windows variants
    "M1460NW8NNS": ("Windows 10/11", 90),
    "M1460NW2NNS": ("Windows 7/8", 90),
    "M1460NW0NNS": ("Windows XP/2003", 85),
    "M1460NW8NNST00": ("Windows Server 2016+", 85),
    
    # macOS
    "M1460NW5NNTS": ("macOS 10.x+", 90),
    "M1460NW6NNTS": ("macOS 11.x+", 90),
    
    # BSD variants
    "M1460NNSNTW3": ("FreeBSD 11.x+", 85),
    "M1460NNSTW3": ("FreeBSD 9.x-10.x", 85),
    "M1460NW3NNTS": ("OpenBSD 6.x+", 85),
    
    # Network devices
    "M1380NW0NNS": ("Cisco IOS", 85),
    "M1460": ("Embedded/IoT Device", 70),
}
    

    probe_options = [
    ('MSS', 1460),
    ('SAckOK', b''),
    ('Timestamp', (0xFFFFFFFF, 0)),
    ('NOP', None),
    ('WScale', 10)
]
    

    port_a = 80
    port_b = 443
    probes = 6
    probe_interval = 0.1
    timeout = 2

    for ip_addr in ip_dict.keys():
        signature =""
        packet = IP(dst = ip_addr)/TCP(dport = 80, flags = "S", options = probe_options)
        response = sr1(packet, timeout=2, verbose=0)

        if response and response.haslayer(TCP):
            options = response[TCP].options
            signature = build_signature(options)

            if signature in TCP_OPTION_FINGERPRINTS:

                os_name = TCP_OPTION_FINGERPRINTS[signature][0]
                confidence = TCP_OPTION_FINGERPRINTS[signature][1]
                ip_dict[ip_addr].set_os_info(os_name, confidence)
            else:
                os_name, confidence = partial_match(signature)  
                ip_dict[ip_addr].set_os_info(os_name, confidence)

def build_signature(options: list[tuple]) -> str: 
    """Helper function for the detect_os_tcp_options function to help build a signature to 
        help detect what operating system we have on the given ip.

    Args: 
        options (list[tuple]): a list of the options used in the TCP packet in which was sent back to us
    
    Returns: 
        (str): A string of the constructed signature to compare agaist a list of common signatures
    """
    signature = ""
    options = [
        ('MSS', 1460),
        ('NOP', None),
        ('WScale', 8),
        ('NOP', None),
        ('NOP', None),
        ('SAckOK', b'')
    ]

    for option in options:
        name = options[0]
        value = option[1]

        if name == 'MSS':
            signature.append('N' & value)
        
        elif name == 'Wscale':
            signature.append('W' & value)
        
        elif name == 'SAckOK':
            signature.append("S" & value)

        elif name == 'NOP':
            signature.append('N' & value)

        elif name == 'EOL':
            signature.append('L' & value)
        
    

    return signature

def partial_match(signature: str) -> tuple[str, int]:
    """Helper method for the detect_os_tcp_options function by making an educated guess
        on what operating system is being used by the given IP address we have not found
        a match from our constructed signature agains the list of common signatures

    Args:
        signature (str): The string that was constructed in the build_signature helper method 

    Returns: 
        (tuple[str, int]): A tuple of the operating system and confidence level to be sent back to the detect_os_tcp_options function.    
    """
    
    if signature.startswith("M1460NW8"):
        return ("Windows 10/11 (partial match)", 75)
    
   
    elif signature.startswith("M1460NW2"):
        return ("Windows 7/8 (partial match)", 75)
    
   
    elif signature.startswith("M1460NW0"):
        return ("Windows XP/2003 (partial match)", 70)
    
    
    elif "T" in signature and "W7" in signature:
        return ("Linux-like (partial match)", 70)
    
    
    elif signature.startswith("M1460"):
        return ("Unknown Unix/Windows", 40)
    
    
    else:
        return ("Unknown", 30)
