def service_scan(ip: dict[str, Address_Info],
               timeout: int | float = 5) -> None:
    """Attempts to identify the actual service running on a port. 
    Adds the data to a dictionary in the Address_Info object

    Services that can be detected:
    - SSH (Any that follow RFC 4253 and those that don't)
    - FTP (vsFTPd, FileZilla and others)
    - HTTP
    - Telnet (Only inetutils version)
    - MySQL (Most likely only MariaDB)
    - SMTP

    Args:
        ip (dict[str, Address_Info]): The dictionary of Address_Info objects to scan
        timeout (int | float = 5): The timeout for each socket
    """
    ips = list(ip.keys())

    socket.setdefaulttimeout(timeout) # Sets the timeout for each socket
    MAX = os.cpu_count() * MULTIPLIER # Most likely a safe amount of threads

    cur_threads = set() # The list of current threads
    
    done = set()
    with concurrent.futures.ThreadPoolExecutor(max_workers=MAX) as executor:
        for i in ips:
            for port in ip[i].oports:
                cur_threads.add(executor.submit(worker, ip=i, port=port))

                if len(cur_threads) >= MAX:
                    done, cur_threads = concurrent.futures.wait(cur_threads, 
                                                                return_when=concurrent.futures.FIRST_COMPLETED)
                    
        done, cur_threads = concurrent.futures.wait(cur_threads)
            
    for future in done:
        ip_addr, port, service = future.result()
        if service:
            ip[ip_addr].services[port] = service

def worker(ip: str, port: int) -> tuple[str, int, str]:
    """The worker thread for multithreading purposes. Splits off each socket pair into their 
    own thread to speed up scanning.

    Args:
        ip (str): The IP address to scan
        str (int): The port to scan
    
    Returns:
        (tuple[str, int, str]): The IP, port, and identifying service information in that order
    """
    ip_port = (ip, port)

    null_scan_soc = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    get_scan_soc = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    
    service = ""

    try: # Null scan
        null_scan_soc.connect(ip_port)
        data = null_scan_soc.recv(1024)
        null_scan_soc.close()
        service = identify(data)
    except (TimeoutError, ConnectionRefusedError, ConnectionResetError) as e:
        try:
            get_req = b"GET / HTTP/1.1\r\nConnection: close\r\n\r\n"
            get_scan_soc.connect(ip_port)
            get_scan_soc.send(get_req)
            data = get_scan_soc.recv(1024)
            get_scan_soc.close()
            service = identify(data)
        except (TimeoutError, ConnectionRefusedError, ConnectionResetError) as e:
            pass            
    
    return ip, port, service
