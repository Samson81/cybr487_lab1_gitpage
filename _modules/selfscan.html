def get_sockets_in_proc(protocol: str) -> list[dict]:
    """Parse /proc/net/tcp or /proc/net/udp file
    
    Args:
        protocol (str): tcp or udp protocol, which one to search

    Returns:
        (list[dict]): list of socket info dictionaries
    """
    filepath = f"/proc/net/{protocol}"
    sockets = []
    
    try:
        with open(filepath, 'r') as f:
            #no header pls
            next(f)
            
            for line in f:
                parts = line.split()
                if len(parts) < 10:
                    continue
                
                #get fields
                local_address = parts[1]
                remote_address = parts[2]
                state = parts[3]
                inode = parts[9]
                
                #get addresses from hex
                addr_half, port_half = local_address.split(':')
                local_port = int(port_half, 16)
                addr_int = int(addr_half, 16)
                addr_int = socket.ntohl(addr_int)
                local_ip = socket.inet_ntoa(struct.pack('!I', addr_int))
                
                socket_info = {
                    'local_ip': local_ip,
                    'local_port': local_port,
                    'state': state,
                    'inode': inode,
                    'protocol': 'tcp' if 'tcp' in filepath else 'udp'
                }
                
                sockets.append(socket_info)
    
    except (IOError, OSError) as e:
        print(f"Error reading {filepath}: {e}")
    
    return sockets

def get_process_info(inode: str) -> Tuple[Optional[str], Optional[str]]:
    """Find which process owns a socket by matching inode in /proc/[pid]/fd/

    Args:
        inode (str): The inode for the process
    Returns: 
        (Tuple[Optional[str], Optional[str]]): (pid, process_name) or (None, None)
    """
    for pid in os.listdir('/proc'):
        if not pid.isdigit():
            continue
        
        try:
            for fd in os.listdir(f'/proc/{pid}/fd'):
                try:
                    link_target = os.readlink(f'/proc/{pid}/fd/{fd}')
                    
                    if link_target == f'socket:[{inode}]':
                        try:
                            with open(f'/proc/{pid}/comm', 'r') as f:
                                return pid, f.read().strip()
                        except:
                            return pid, "unknown"
                
                except (OSError, IOError):
                    continue
        
        except (OSError, IOError):
            continue
    
    return None, None

def is_listening_socket(socket_info: dict[str, str]) -> bool:
    """Determine if socket is in listening state
    TCP state 0A = listening (10 in decimal)

    Args:
        socket_info (dict[str, str]): The information about the socket

    Returns:
        bool: Whether the socket is listening or not
    """
    if socket_info['protocol'] == 'tcp':
        #state 0A = 10 = listen socket
        return socket_info['state'] == '0A'
    else:
        return True

def scan_self() -> list[dict]:
    """Scan for all listening ports on the system using /proc

    Returns: 
        list[dict]: List of dictionaries with port information
    """
    if not os.path.exists('/proc'):
        raise RuntimeError("Self scan not available without a /proc filesystem.")
    
    listening_ports = []
    
    #get all tcp and udp sockets then combined them
    tcp_sockets = get_sockets_in_proc('tcp')
    udp_sockets = get_sockets_in_proc('udp')
    all_sockets = tcp_sockets + udp_sockets
    
    #filter just listening sockets
    for sock in all_sockets:
        if is_listening_socket(sock):
            pid, process_name = get_process_info(sock['inode'])
            
            port_info = {
                'port': sock['local_port'],
                'protocol': sock['protocol'].upper(),
                'address': sock['local_ip'],
                'pid': pid,
                'process': process_name
            }
            
            listening_ports.append(port_info)
    
    return listening_ports
