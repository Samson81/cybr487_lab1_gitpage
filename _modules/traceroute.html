def traceroute(target_info: Address_Info.Address_Info,
               args: argparse.Namespace) -> None:
    """Perform traceroute to target IP using ICMP or TCP
    
    Args:
        target_info: Address_Info object to populate with hop data
        args (argparse.Namespace): The arguments passed into the analyzer

    """
    if args.traceroute.lower() == 'icmp':
        hops = icmp_traceroute(target_info.ip(), args.timeout, args.traceroutehops)
    elif args.traceroute.lower() == 'tcp':
        hops = tcp_traceroute(target_info.ip(), args.timeout, args.traceroutehops)
    
    target_info.hops = hops

def icmp_traceroute(target_ip: str,
                    timeout: int,
                    max_hops: int) -> list[str]:
    """Perform ICMP-based traceroute
    
    Uses ICMP Echo Request packets with incrementing TTL values.
    Each hop is probed 3 times to handle packet loss.
    
    Args:
        target_ip (str): Destination IP address
        timeout (int): Timeout for each probe
        max_hops (int): Maximum hops to try
        
    Returns:
        list[str]: Formatted hop information
    """
    hops = []
    reached_destination = False
    
    for ttl in range(1, max_hops + 1):
        hop_info = probe_hop_icmp(target_ip, ttl, timeout)
        
        if hop_info['ip'] is None:
            hops.append(f"{hop_info['ip_display']} ({hop_info['avg_rtt']})")
        else:
            hops.append(f"{hop_info['ip']} ({hop_info['avg_rtt']})")
            
        print(f"     {ttl:02}. {hops[-1]}")
            
        #cjeck if we reached the destination
        if hop_info['ip'] == target_ip:
            reached_destination = True
            break
    
    if not reached_destination and ttl == max_hops:
        hops.append(f"Max hops ({max_hops}) reached - destination may be further")
    
    return hops

def tcp_traceroute(target_ip: str,
                   timeout: int,
                   max_hops: float,
                   port: int = 80) -> list[str]:
    """Perform TCP-based traceroute
    
    Uses TCP SYN packets with incrementing TTL values.
    Each hop is probed 3 times to handle packet loss.
    
    Args:
        target_ip (str): Destination IP address
        timeout (int): Timeout for each probe
        max_hops (int): Maximum hops to try
        port (int): Target port (default 80)
        
    Returns:
        list[str]: Formatted hop information
    """
    hops = []
    reached_destination = False
    
    for ttl in range(1, max_hops + 1):
        hop_info = probe_hop_tcp(target_ip, ttl, timeout, port)
        
        if hop_info['ip'] is None:
            hops.append(f"{hop_info['ip_display']} ({hop_info['avg_rtt']})")
        else:
            hops.append(f"{hop_info['ip']} ({hop_info['avg_rtt']})")

        print(f"     {ttl:02}. {hops[-1]}")
            
        #check if we reached the destination
        if hop_info['ip'] == target_ip:
            reached_destination = True
            break
    
    if not reached_destination and ttl == max_hops:
        hops.append(f"Max hops ({max_hops}) reached - destination may be further")
    
    return hops

def probe_hop_icmp(target_ip: str,
                   ttl: int,
                   timeout: int,
                   attempts: int = 3) -> dict[str, str]:
    """Probe a single hop using ICMP, attempting 3 times
    
    Args:
        target_ip (str): Destination IP
        ttl (int): Time-to-live value for this hop
        timeout (int): Timeout for probe
        attempts (int): Number of attempts per hop (default 3)
        
    Returns:
        (dict[str, str]): Contains 'ip', 'ip_display', 'avg_rtt', 'responses'
    """
    responses = []
    rtt_times = []
    responding_ip = None
    
    for attempt in range(attempts):
        #create ICMP Echo Request with specific TTL
        packet = IP(dst=target_ip, ttl=ttl) / ICMP()
        
        #send packet and wait for response
        start_time = time.time()
        reply = sr1(packet, timeout=timeout, verbose=0)
        end_time = time.time()
        
        if reply is not None:
            rtt = (end_time - start_time) * 1000  #milliseconds probably better
            rtt_times.append(rtt)
            
            #get the response IP only if we got ICMP response
            if reply.haslayer(ICMP):
                if responding_ip is None:
                    responding_ip = reply.src
                responses.append(True)
            else:
                responses.append(False)
        else:
            responses.append(False)
    
    #calc results
    successful_probes = sum(responses)
    
    if successful_probes == 0:
        #all packets dropped
        return {
            'ip': None,
            'ip_display': '* * *',
            'avg_rtt': 'Request timed out',
            'responses': responses
        }
    else:
        avg_rtt = sum(rtt_times) / len(rtt_times)
        return {
            'ip': responding_ip,
            'ip_display': responding_ip,
            'avg_rtt': f"{avg_rtt:.2f} ms",
            'responses': responses
        }

def probe_hop_tcp(target_ip: str,
                  ttl: int,
                  timeout: int,
                  port: int,
                  attempts: int = 3) -> dict[str, str]:
    """Probe a single hop using TCP SYN, attempting 3 times
    
    Args:
        target_ip (str): Destination IP
        ttl (int): Time-to-live value for this hop
        timeout (int): Timeout for probe
        port (int): Target TCP port
        attempts (int = 3): Number of attempts per hop (default 3)
        
    Returns:
        (dict[str, str]): Contains 'ip', 'ip_display', 'avg_rtt', 'responses'
    """
    responses = []
    rtt_times = []
    responding_ip = None
    
    for attempt in range(attempts):
        #create TCP SYN packet with specific TTL
        packet = IP(dst=target_ip, ttl=ttl) / TCP(dport=port, flags="S")
        
        #send packet and wait for response
        start_time = time.time()
        reply = sr1(packet, timeout=timeout, verbose=0)
        end_time = time.time()
        
        if reply is not None:
            rtt = (end_time - start_time) * 1000  #milliseconds
            rtt_times.append(rtt)
            
            #get the response IP only if we get ICMP time exceeded or
            #TCP response from final destination
            if reply.haslayer(ICMP) or reply.haslayer(TCP):
                if responding_ip is None:
                    responding_ip = reply.src        
                responses.append(True)
            else:
                responses.append(False)
        else:
            responses.append(False)
    
    #calc results
    successful_probes = sum(responses)
    
    if successful_probes == 0:
        #all packets dropped
        return {
            'ip': None,
            'ip_display': '* * *',
            'avg_rtt': 'Request timed out',
            'responses': responses
        }
    else:
        avg_rtt = sum(rtt_times) / len(rtt_times)
        return {
            'ip': responding_ip,
            'ip_display': responding_ip,
            'avg_rtt': f"{avg_rtt:.2f} ms",
            'responses': responses
        }
