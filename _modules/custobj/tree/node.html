class Node:
    """The Node class for the tree
    """
    def __init__(self, ip_addr: Hop, count: int, hops: list[Hop] | None = None) -> None:
        """Creates a node for the traceroute tree for visualization

        Args:
            ip_addr (str): The IP address for the current Node
            hops (list[str]): The hops past this IP address in the traceroute
            count (int): The depth of the tree to this node (Used to distinguish (* * *))
        """
        if isinstance(ip_addr, Hop):
            self.ip = ip_addr.hop # The IP address
            self.rtt = ip_addr.rtt # Round Trip Time
        else:
            self.ip = ip_addr # The user's IP address
            self.rtt = 0
        
        self.hops = {} # A dictionary of the unique hops past this IP address
        self.depth = count # The depth in the tree. Used to distinguish (* * *) points
        
        if hops:
            self._parse_info(hops, count + 1)
    
    def __str__(self):
        return f"IP: {self.ip}\nRTT: {self.rtt}\nHops: {self.hops}\nDepth: {self.depth}"

    def _parse_info(self, hops: list[Hop], count: int) -> None:
        """Extracts the information in the Address_Info object, and build this tree

        Args:
            hops (list[str]): The list of hops past this Node
            count (int): The depth of the next Node
        """
        if len(hops) == 0:
            return
        next_node = hops.pop(0)

        if next_node in self.hops.keys():
            self.hops[next_node.hop]._parse_info(hops, count)
        else:
            self.hops[next_node.hop] = Node(next_node, count, hops)
