"""Tree for visualization

from custobj.Address_Info import Address_Info
from custobj.tree.node import Node
from socket import gethostbyname_ex, gethostname
from custobj.hop import Hop

class Tree:
    """The Tree object to make visualizing the network easier
    """
    def __init__(self, network_map: dict[str, Address_Info]):
        """Initializes the Tree object

        Args:
            network_map (dict[str, Address_Info]): The network map dictionary that contains all traceroute hops
        """
        host_data = gethostbyname_ex(gethostname()) # The IP address of your machine
        if(len(host_data[2]) >= 2):
            temp_list = host_data[len(host_data) - 1] # Should be the list of IP addresses
            local_ip = temp_list[len(temp_list) - 1] # Might be the actual for the machine :shrug:, works on mine ;)
            # It ends up being one of them
        else:
            local_ip = '127.0.0.1'
        self.root = Node(local_ip, 0)

        hop_list = self._parse_addr_info(network_map)

        for traceroute in hop_list:
            self.root.add_path(traceroute)
        

    def _parse_addr_info(self, network_map: dict[str, Address_Info]) -> list[list[Hop]]:
        """Creates a list of list of Hops object representing each traceroute, and returns it

        Args:
            network_map (dict[str, Address_Info]): The network_map that gets filled out by other functions

        Returns:
            (list[list[Hop]]): A list of all the traceroutes (which are lists of individual Hops)
        """
        hop_list = []
        for key in network_map.keys():
            info = network_map[key]
            hop_list.append(info.hops())
        return hop_list
