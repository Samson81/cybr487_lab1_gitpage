class Address_Info:
    """A Class for tracking information about an IP address

    - Hops from this IP
    - Open ports
    - Closed ports
    """
    def __init__(self, ipaddr: str) -> None:
        """Creates a new Address_Info with the given IP, and initializes the object

        Args:
            ipaddr (str): The IP address for this object
        """
        if not ipaddr:
            self._ip = '0.0.0.0'
        else:
            self._ip = ipaddr 

        self._hops = [] # List of hop objects
        self.oports = [] # Open scanned ports
        self.cports = [] # Open closed ports
        self.services = {} # The services running on each port {port: service} Use as alternative to oports
        self.os_name = None          # Add these
        self.os_confidence = 0       # Add these

    def set_os_info(self, os_name: str, confidence: int) -> None:
        """Update OS detection info if confidence is higher
        Args:
            os_name (str): Detected OS name
            confidence (int): Confidence level (0-100)
        """
        if confidence > self.os_confidence:
            self.os_name = os_name
            self.os_confidence = confidence

    def get_os_info(self) -> tuple[str, int]:
        """Returns the detected OS and confidence
        Returns:
            tuple[str, int]: (os_name, confidence)
        """
        return (self.os_name, self.os_confidence)



    def __str__(self):
        return f"{self.ip()}:\nTraceroute Hops: {self._hops}\nOpen Ports: {self.oports}\nClosed Ports:p {self.cports}"

    def add_hop(self, hop: str, rtt: float) -> None:
        """Adds a hop with it's round trip time to the list

        Args:
            hop (str): The IP address of the hop
            rtt (float): The Round Trip time for the packet
        """
        self._hops.append(Hop(hop, rtt))

    def hops(self) -> list[Hop]:
        """Returns the list of hops and round trip times

    def scanned_port(self, port: int, open: bool) -> None:
        """Appends to a list whether the scanned port was open or closed.

        Args:
            port (int): The port to append
            open (bool): Whether the port was open or closed
        """
        if open:
            self.oports.append(port)
        else:
            self.cports.append(port)
