def visualizer(network_map: dict[str, Address_Info]) -> None:
    """Visualizes the network based on the traceroutes to each IP

    Args:
        network_map (dict[str, Address_Info]): A dictionary containing the filled out data for each IP address scanned
    """
    
    tree = Tree(network_map)
    
    queue = []
    queue.append(tree.root)
    root_node = pydot.Node(f"{tree.root.ip}{tree.root.depth}", label=tree.root.ip, shape="circle", style="filled", fillcolor="lightblue")

    # Kind of written by AI
    # AI was used to make the graph appear better
    # Prompt (kind of): <code> I just want the nodes to spread evenly around the root node
    graph = pydot.Dot(graph_type ="graph", bgcolor="white", layout="circo")
    graph.set_overlap("false")
    graph.set_splines("curved")
    graph.set_ranksep("2")
    graph.set_nodesep("1")

    
    graph.add_node(root_node)
    
    while len(queue) > 0:
        tree_node = queue.pop(0)
        for hop in tree_node.hops.keys():
            tree_hop = tree_node.hops[hop]
            if tree_hop.rtt == -1 or isinstance(tree_hop.rtt, str):
                rtt = ""
            else:
                rtt = f"{float(tree_hop.rtt):.2f} ms"
            
            pnode = pydot.Node(f"{tree_hop.ip}{tree_hop.depth}", label=f"{tree_hop.ip}\n{rtt}", shape="circle", style="filled", fillcolor="lightblue", rank=str(tree_hop.depth))
            graph.add_node(pnode)
            
            graph.add_edge(pydot.Edge(f"{tree_node.ip}{tree_node.depth}", f"{tree_hop.ip}{tree_hop.depth}", penwidth='2.0', labeldistance="2.5", labelangle="25",))
            queue.append(tree_hop)

    graph.write_pdf('output/visualization.pdf')
