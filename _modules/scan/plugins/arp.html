def scan(ip: str | list[str],
         timeout: int = 2) -> list[str]:
    """Performs an ARP packet scan on the provided IP range.

    Args:
        ip (str | list[str]): A string or list that contain the IP addresses to check
        timeout (int): The time in seconds to wait for a packet

    Returns:
        str: A list of the IP addresses that responded
    """
    conf.verb = 0

    eth_packet = Ether()
    arp_packet = ARP()

    eth_packet.dst = 'ff:ff:ff:ff:ff:ff'
    arp_packet.pdst = ip
    arp_packet.op = 1

    packet = eth_packet/arp_packet

    ans, unans = srp(packet, timeout=timeout, verbose=False)
    seen_ips = []

    for pack in ans:
        seen_ips.append(pack.answer[ARP].psrc) # The source IP for the returned 'is-at' packet

    return seen_ips
