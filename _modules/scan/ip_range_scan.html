"""Scan class

Provides:
 - the default Scan class
 - a simple interface for all scan types
 - scan() method
"""

import pkgutil
import importlib
import scan.plugins as scans
import ipaddress
from threads.CustomThread import CustomThread, loading, NOT_DONE

class Scan:
    def __init__(self,
                 ip_range: str | list[str],
                 timeout : int | float = 2,
                 scan_type : str = "arp") -> None:
        """Initialization of the Scan class

        Args:
            ip_range (str | list[str]): A string or list that represents the range of IP addresses in CIDR notation.
            timeout (int | float): An integer or float that represents the time in seconds to wait for a packet.
            scan_type (str): A string that indicates what scan to run. (should be defined in the plugin)
        """
        self._ip_range = ip_range
        self._timeout = timeout
        self._scan_type = scan_type
        self._seen_ips = []

        self._plugins = self._load_plugins()

    def _load_plugins(self) -> dict:
        """Loads the plugins located in scan/plugins

        Plugins that come by default:
        - arp
        - icmp
        - nmap
        - tcp

        Returns:
            dict: A dictionary of all the loaded plugins.
        """
        plugins = {}

        for plugin in pkgutil.iter_modules(scans.__path__):
            module = importlib.import_module(f"{scans.__name__}.{plugin.name}")

            scan_name = getattr(module, "SCAN_TYPE", "")
            scan_function = getattr(module, "scan", "")

            if scan_name != "" and callable(scan_function):
                plugins[scan_name] = scan_function
            else:
                print(f"Invalid Plugin Loaded: {plugin.name}\nSkipping!!!")

        return plugins

    def scan(self) -> set[str]:
        """Calls the chosen scan types scan function with the IP range

        Scans the given IP range

        Returns:
            set[str]: The set of IP addresses that responded, sorted
        """
        scan_function = self._plugins.get(self._scan_type)

        if not scan_function:
            print(f"Undefined Scan Type {self._scan_type}\n Using ARP.")
            scan_function = self._plugins.get(self._scan_type)

        t1 = CustomThread(scan_function, ip=self._ip_range, timeout=self._timeout)
        t2 = CustomThread(loading, not_done=NOT_DONE) # Creates a loading spinner

        t1.start()
        t2.start()

        t1.join()
        NOT_DONE.set() # Set event for loading bar
        t2.join()

        self._seen_ips = t1.result

        # Sorts the IP addresses and removes duplicates
        temp_list = set(sorted(self._seen_ips, key=ipaddress.IPv4Address))

        return temp_list
