class CustomThread(threading.Thread):
    """A custom wrapper for threading.Thread that allows for
     obtaining the return value of the called function
    """
    def __init__(self, target: callable, **kwargs: dict) -> None:
        """The init function for CustomThread

        Args:
            target (function): The function to run
            kwargs (dict): The keyword arguments to pass into the function
        """
        super().__init__()
        if kwargs is None:
            kwargs = {}

        self._target = target
        self._kwargs = kwargs
        self.result = []

def loading(not_done: threading.Event) -> None:
    """A simple loading bar to show the program is still running

    Args:
        not_done (threading.Event): The event to stop the loading bar
    """
    spinners = ["|", "/", "-", "\\"]
    for c in itertools.cycle(spinners):
        print(f"\r{c}", end="")
        if not_done.is_set():
            break
        time.sleep(0.2)
