def host_report(idx: int, target_ip: str, colors: dict[str, str]) -> None:
    """Prints the host IP address in a box

    Args:
        idx (int): The iterator number to display
        target_ip (str): The IP address to display
        colors (dict[str, str]): The dictionary of colors for display purposes
    """
    host_text = f"{idx:02d} │ Host: {target_ip}"
    box_width = len(host_text) + 4
    host_text = f"{colors['BOLD']}{idx:02d} │ Host: {target_ip}{colors['RESET']}"

    print(f"\n╭{'─' * (box_width - 2)}╮")
    print(f"│ {host_text} │")
    print(f"╰┬{'─' * (box_width - 3)}╯")
