def portscan_report(target_info: Address_Info,
                    do_traceroute: bool,
                    args: Namespace,
                    colors: dict[str,str])-> None:
    """Displays the information from the portscan, as well as displaying CVE information

    Args:
        target_info (Address_Info): The object containing information about an IP address
        do_traceroute (bool): Whether to do a traceroute (required for some nice output)
        args (Namespace): The args passed into the main function
        colors (dict[str, str]): The dictionary of colors for display purposes
    """
    #draw scan name
    ps_name = 'Port'
    if args.service_scan:
        ps_name += r'/Service'
    if args.cve:
        ps_name += r'/CVE'
    ps_name += ' Scan'
    box_width = len(ps_name)
    
    print(f" │")
    prefix = ' ├─' if do_traceroute else ' ╰─'
    print(f" │ ╭─{'─' * (box_width)}─╮")
    print(f"{prefix}┤ {colors['BOLD']}{ps_name}{colors['RESET']} │")
    prefix = ' │ ' if do_traceroute else '   '
    print(f"{prefix}╰┬─{'─' * (box_width - 1)}─╯")
    
    if target_info.oports:
        ports = sorted(target_info.oports)
        for i, port in enumerate(ports):
            if Port_Parse.all:
                service = Port_Parse.get_service(port) 
            else:
                # If the services have not been filled
                # out in Port_Parse for some reason
                service = COMMON_PORTS.get(port, "Unkown")
            
            if args.service_scan:
                if port in target_info.services.keys():
                    service = target_info.services[port]
            
            prefix2 = ' ╰─' if i == len(ports) - 1 else ' ├─'
            print(f"{prefix}{prefix2}{colors['BOLD']}{colors['GREEN']}[Port {port}/tcp: OPEN]{colors['RESET']} ({service})")
            
            if args.cve and service and args.service_scan:
                prefix2 = '    ' if i == len(ports) - 1 else ' │  '
                indent = f"{prefix}{prefix2}"
                
                cve_info = search_exploitdb(service)
                
                for cve in cve_info:
                    print(f"{indent}‣ {colors['RED']}CVE #{cve['id']}{colors['RESET']} - "
                            f"{cve['platform']} {cve['type']} vulnerability")
                            
                    text = f"    {cve['description']}"
                    wrapped = textwrap.fill(text, width=80,
                        initial_indent=indent,
                        subsequent_indent=f"{indent}    ")
                    print(f"{wrapped}")
                    
                if i != len(ports) - 1 and cve_info:
                    print(indent)
    else:
        print(f"{prefix} ╰─{colors['RED']}No Scanned Ports Open (Closed or Filtered){colors['RESET']}")
