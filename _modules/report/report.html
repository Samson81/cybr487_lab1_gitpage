"""Display detailed scan results for discovered hosts
"""

from argparse import Namespace
from custobj.Address_Info import Address_Info
from custom.threads.CustomThread import THREAD_DONE
import time
from report.all import *


colors = {
'BOLD': '\033[1m',
'YELLOW': '\033[93m',
'RED': '\033[91m',
'GREEN': '\033[92m',
'RESET': '\033[0m'
}

def netscan_report(network_map: dict[str, Address_Info],
                   args: Namespace) -> None:
    """Display detailed information for each discovered IP
    
    Args:
        network_map (dict[str, Address_Info.Address_Info]): The list of found IP addresses
        args (Namespace): The arguments passed into the analyzer
    """
    
    do_traceroute = args.traceroute is not None

    #sort IPs numerically by converting each octet to an integer
    sorted_ips = sorted(
        network_map.items(),
        key=lambda x: [int(part) for part in x[0].split('.')]
    )

    THREAD_DONE.set() # Set event for loading bar to stop
    time.sleep(1) # Make sure it finishes

    if len(sorted_ips) < 1:
        print(f"\r\n╭{"─" * 54}╮") # 54 is the max length of the below strings
        print(f"│ {colors['RED']}No Scanned IP Addresses Responded{colors['RESET']}{" " * 19} │\n\
│ {colors['RED']}Host(s) may be down, or are ignoring the chosen scan{colors['RESET']} │")
        print(f"╰{"─" * 54}╯")

    for idx, (target_ip, target_info) in enumerate(sorted_ips, 1):
        host_report(idx, target_ip, colors)
        
        #report on optional os detection
        if args.os:
            os_report(do_traceroute, args.portscan, target_info, colors)
        
        #report on optional port scan
        if args.portscan is not None:
            portscan_report(target_info, do_traceroute, args, colors)
            
        
        #do the traceroute if requested
        if do_traceroute:
            traceroute_report(args, target_info, colors)
                
    print()
